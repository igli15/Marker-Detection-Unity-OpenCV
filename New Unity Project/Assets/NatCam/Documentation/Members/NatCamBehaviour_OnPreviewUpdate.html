NatCamBehaviour.OnPreviewUpdate

public void OnPreviewUpdate ();

Description

Method called on every new camera preview frame. 

public class PreviewFramerate : NatCamBehaviour {
    
    public Text text; //UI text to show frame rate

    private float lastTime, delta, framerate = 30f;

    //OnPreviewUpdate is called once per preview frame
    public override void OnPreviewUpdate () {
        //Calculate time delta
        delta = Time.unscaledTime - lastTime;
        //Check that delta isn't zero
        if (delta == 0f) return;
        //Calculate the framerate
        framerate = 1f / delta;
        //Display the framerate
        text.text = framerate.ToString("F2");
        //Update last time
        lastTime = Time.unscaledTime;
    }
}
