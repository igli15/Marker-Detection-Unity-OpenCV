NatCam.PreviewFrame

public static bool PreviewFrame (Texture2D& frame);

Description

Get the current preview frame. Note that you must free the texture once you are done with it by using Texture2D.Destroy(Texture2D). 

void SaveScreenshot () {
    // Declare the frame texture
    Texture2D frame;
    // Get the preview frame
    NatCam.PreviewFrame(ref frame);
    // Encode it to JPEG
    byte[] jpg = frame.EncodeToJPG();
    // Save it
    System.IO.WriteAllBytes("screenshot.jpg", jpg);
    // Free the texture and its memory // Note that this is very important to avoid a memory leak
    Texture2D.Destroy(frame);
}
