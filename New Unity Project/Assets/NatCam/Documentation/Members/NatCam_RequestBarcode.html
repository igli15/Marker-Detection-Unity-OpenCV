NatCam.RequestBarcode

public static void RequestBarcode (MetadataCallback callback, BarcodeFormat format, bool detectOnce);

Description

Request to be notified when a barcode is detected. 

void Start () {
    // Start the camera preview
    NatCam.Play(DeviceCamera.RearCamera);
    // Create an ISBN request // You can request multiple barcode formats by using bitwise OR
    NatCam.RequestBarcode(new BarcodeRequest(OnDetectedISBN, BarcodeFormat.EAN_13 | BarcodeFormat.EAN_8, false));
}

void OnDetectedISBN (Barcode code) {
    // Log what we found
    Debug.Log("Found ISBN => " + code.ToString());
}
