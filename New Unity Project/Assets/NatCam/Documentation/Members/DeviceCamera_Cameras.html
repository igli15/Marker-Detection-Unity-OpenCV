DeviceCamera.Cameras

public static ReadOnlyCollection<DeviceCamera> Cameras;

Description

A readonly list of all cameras on the device. 

// Switch on camera torches
for (int i = 0, len = DeviceCamera.Cameras.Count; i < len; i++) {
    // Cache the camera
    DeviceCamera camera = DeviceCamera.Cameras[i];
    // Check if it supports torch
    if (camera.IsTorchSupported) {
        // Set it as the active camera
        NatCam.Camera = camera;
        // Put on the torch
        camera.TorchMode = Switch.On;
        // Terminate the loop
        break;
    }
}
